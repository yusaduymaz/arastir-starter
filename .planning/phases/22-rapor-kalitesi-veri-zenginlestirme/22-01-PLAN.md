    ```
    function getLatestNonNull(data: any[], field: string): string | null {
      if (!data || data.length === 0) return null;
      for (let i = data.length - 1; i >= 0; i--) {
        const val = data[i][field];
        if (val != null && val !== '' && val !== 'ND') return String(val);
      }
      return null;
    }
    ```
